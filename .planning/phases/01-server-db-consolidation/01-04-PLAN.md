```bash
ssh -i "/c/Users/maxfe/Maxflow Software/Projekte/Stiftung TimeTracker/TimeTracking-Clean/.ssh/oracle_server.key" \
  -o StrictHostKeyChecking=no \
  ubuntu@129.159.8.19 \
  "SOURCE_DB='/home/ubuntu/database-shared.db' && \
   TIMESTAMP=\$(date +%Y%m%d_%H%M%S) && \
   BACKUP_PATH=\"/home/ubuntu/databases/backups/production.\${TIMESTAMP}.db\" && \
   \
   echo '--- Pre-backup: source file state ---' && \
   ls -lh \"\$SOURCE_DB\" && \
   \
   echo \"--- Creating backup: \$BACKUP_PATH ---\" && \
   cp \"\$SOURCE_DB\" \"\$BACKUP_PATH\" && \
   \
   echo '--- Setting permissions on backup ---' && \
   chmod 600 \"\$BACKUP_PATH\" && \
   chown ubuntu:ubuntu \"\$BACKUP_PATH\" && \
   \
   echo '--- Post-backup: verify backup file ---' && \
   ls -lh \"\$BACKUP_PATH\" && \
   \
   echo '--- Size comparison ---' && \
   SRC_SIZE=\$(stat -c '%s' \"\$SOURCE_DB\") && \
   BAK_SIZE=\$(stat -c '%s' \"\$BACKUP_PATH\") && \
   echo \"Source size:  \$SRC_SIZE bytes\" && \
   echo \"Backup size:  \$BAK_SIZE bytes\" && \
   [ \"\$SRC_SIZE\" = \"\$BAK_SIZE\" ] && echo 'SIZE_MATCH: OK' || echo 'SIZE_MATCH: FAIL' && \
   \
   echo '--- Original still present ---' && \
   ls -lh \"\$SOURCE_DB\" && \
   \
   echo '--- All backups ---' && \
   ls -lh /home/ubuntu/databases/backups/"
```

SAFETY CHECK: If output shows `SIZE_MATCH: FAIL`, the backup copy was not clean. Investigate before proceeding.
